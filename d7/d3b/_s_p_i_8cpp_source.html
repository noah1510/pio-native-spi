177 void SPIClass::transfer(void *buf, uint16_t count) {
178  uint8_t *cbuf = reinterpret_cast<uint8_t*>(buf);
179 
180  // cbuf may not be 32bits-aligned
181  for (; (((unsigned long)cbuf) & 3) && count; cbuf++, count--)
182  *cbuf = transfer(*cbuf);
183 
184  // cbuf is now aligned
185  // count may not be a multiple of 4
186  uint16_t count4 = count & ~3;
187  transferBytes(cbuf, cbuf, count4);
188 
189  // finish the last <4 bytes
190  cbuf += count4;
191  count -= count4;
192  for (; count; cbuf++, count--)
193  *cbuf = transfer(*cbuf);
194 }

223  // Set to 32Bits transfer
